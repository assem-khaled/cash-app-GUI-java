    private void jButton_sign_in2ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButton_sign_in2ActionPerformed

        try 
        {
            Customer db_obj = new Customer();       
            boolean check = db_obj.customer_sign_in(jTextField_id1.getText(), jTextField_password1.getText());
            
            if(check)
            {
                signed_customer_id = Integer.valueOf(jTextField_id1.getText());

                User_services_gui user_sev_obj = new User_services_gui();
                user_sev_obj.setVisible(true);
                user_sev_obj.setTitle("Customer Services");
                jFrame_current_user.dispose();
                this.dispose();
                
            }
            else
            {
                throw new Exception("Wrong password");
            }           
        }
        catch(Exception e)
        {
            JOptionPane.showMessageDialog(this, e, "Error", JOptionPane.ERROR_MESSAGE);
        }
    }//GEN-LAST:event_jButton_sign_in2ActionPerformed

    private void jButton_sign_up1ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButton_sign_up1ActionPerformed
        try
        {
            Customer db_obj = new Customer();
            
            db_obj.customer_sign_up(jTextField_id.getText(), jTextField_password.getText(),
                    jTextField_name.getText(), jTextField_balance.getText()
                    , jTextField_address.getText());
            
            JOptionPane.showMessageDialog(this, "Data saved successfully", "Account Saved", JOptionPane.INFORMATION_MESSAGE);
            jFrame_new_user.dispose();
        }
        catch(Exception e)
        {
            JOptionPane.showMessageDialog(this, e, "Error", JOptionPane.ERROR_MESSAGE);
        }
    }//GEN-LAST:event_jButton_sign_up1ActionPerformed
