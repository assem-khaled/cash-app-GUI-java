    private void TableDisplay_trans_log()
    {
        try
        {
            Transaction db_obj = new Transaction();
            ResultSet rs= db_obj.emp_show_trans_log();

            jTable_trans_log.setModel(DbUtils.resultSetToTableModel(rs));  
        }
        catch(Exception e)
        {
            JOptionPane.showMessageDialog(this, e);
        }
    }
    
    private void TableDisplay_currency()
    {
        try
        {
            Currency db_obj = new Currency();

            jTable_currencies.setModel(DbUtils.resultSetToTableModel(rs));  
        }
        catch(Exception e)
        {
            JOptionPane.showMessageDialog(this, e);
        }
    }
