    private void TableDisplay_currency()
    {
        try
        {
            Currency db_obj = new Currency();
            ResultSet rs= db_obj.show_currency();

        }
        catch(Exception e)

        }
    }

    private void jButton_send_moneyActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButton_send_moneyActionPerformed
        String ret ;
        try
        {
            Transaction db_obj = new Transaction();  
            ret = db_obj.cust_send_money(User_gui.signed_customer_id+"", jTextField_user_id.getText()
                    , jTextField_amount.getText());
            
            if (ret == "Transaction Completed Successfully")
                jFrame_cash_out.dispose();
            
            JOptionPane.showMessageDialog(this, ret, "", JOptionPane.INFORMATION_MESSAGE);

        }
        catch(Exception e)
        {
            JOptionPane.showMessageDialog(this, e, "Error", JOptionPane.ERROR_MESSAGE);
        }
    }//GEN-LAST:event_jButton_send_moneyActionPerformed
